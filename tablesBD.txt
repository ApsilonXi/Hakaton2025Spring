create table users (
	user_login char(20) not null primary key,
	user_password char(20) not null,
	User_role char(15) not null
);

create table tags (
	tag_id serial not null primary key,
	tag_name char(20) not null
);

create table sources (
	source_id serial not null primary key,
	source_name char(20) not null,
	source_link text not null
);

create table news (
	news_id serial not null primary key,
	type_news boolean not null,
	news_title text not null,
	news_content text not null,
	status boolean not null,
	tagID serial not null,
	sourceID serial not null,

	foreign key (tagID) REFERENCES tags (tag_id),
	FOREIGN key (sourceID) REFERENCES sources (source_id)
);

create table folders (
	folder_id serial not null primary key,
	userLOG char(20) not null,
	folder_name char(20) not null,
	newsID serial not null,

	FOREIGN key (userLOG) REFERENCES users (user_login),
	foreign key (newsID) REFERENCES news (news_id)
);
